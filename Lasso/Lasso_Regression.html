def cost(theta_0, theta_1, x, y):
    s = 0
    for i in range(len(x)):
        y_i_hat = x[i]*theta_1 + theta_0
        s += (y[i]-y_i_hat)**2
    return s/len(x)

def cost_lasso(theta_0, theta_1, x, y, lamb):
    s = 0
    for i in range(len(x)):
        y_i_hat = x[i]*theta_1 + theta_0
        s += (y[i]-y_i_hat)**2 + lamb*(abs(theta_0) + abs(theta_1))
    return s/len(x)

def yy(p,soln):
    xx = np.linspace(-soln,soln,100)
    xx_final = []
    yy_final = []
    for x in xx:
        if(x>0):
            xx_final.append(x)
            xx_final.append(x)
            y = (soln**p - x**p)**(1.0/p)
            yy_final.append(y)
            yy_final.append(-y)
            
        else:
            xx_final.append(x)
            xx_final.append(x)
            y = (soln**p - (-x)**p)**(1.0/p)
            yy_final.append(y)
            yy_final.append(-y)
    return xx_final, yy_final
